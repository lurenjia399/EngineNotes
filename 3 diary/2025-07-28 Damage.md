# 1 伤害流程
1 普攻激活 GA_Female051_Melee 这个GA
2 GA_Female051_Melee 这个GA激活 UAbilityTask_PlayMontageAndWaitForEvent 这个GT，这个GT字面意思就是播放Montage，在这个Montage期间等待Event。这个Event是配置在Montage中的 UNotifyState_SendGamepPlayEvent，Event中有Tag。
3 在播放Montage过程中收到了Event，就会调用到GT的 EventReceived 方法广播，进而执行蓝图中 的 OnReceiveEvent 方法
4 蓝图中的 OnReceiveEvent 方法中 会根据Tag找到GA中的 EffectContainerMap 配置，给自己添加Tag对应的GE。 蓝图中的 OnReceiveEvent 方法中 还会根据Tag找到GA中的 SpawnActorMaps 配置，生成Tag对应的UObject
5 在给自己添加GE的流程中，如果是单人模式，会创建一个FHTClientPredictionFightData对象，来保存客户端击中目标的各种信息。
6 会把记录的信息发到服务器来进行校验，是 SendPredictionDamageInfoToServer 这个方法。

USTRUCT()
struct FCharacterForNet
{
	GENERATED_USTRUCT_BODY()

	void Reset();
	
	FServerAISpawnInfo AISpawnInfo;
	FName PlayerCharacterID;
	
	//召唤物索引
	int32 SummonIndex = -1;

	bool NetSerialize(FArchive& Ar, class UPackageMap* Map, bool& bOutSuccess)
	{
		// 定义一个8位的标识位
		uint8 RepBits = 0;
		// 服务器上，往Ar中写入数据
		if (Ar.IsSaving())
		{
			// 如果是PlayerCharacterID，就把1或进标识位
			if (PlayerCharacterID != NAME_None)
			{
				RepBits |= 1<< 0;
			}
			// 如果召唤物索引有值，就在或一个1进标识位
			if (SummonIndex != -1)
			{
				RepBits |= 1 << 1;
			}
		}
		// 因为FCharacterForNet这个结构体只有3个成员变量，这里就序列化3位
		Ar.SerializeBits(&RepBits, 3);
	
		// 如果
		if (RepBits & (1 << 0))
		{
			Ar << PlayerCharacterID;
		}
		else
		{
			AISpawnInfo.NetSerialize(Ar, Map, bOutSuccess);
		}
	
		if (RepBits & (1 << 1))
		{
			Ar << SummonIndex;
		}
		
		return bOutSuccess;
	}
}
