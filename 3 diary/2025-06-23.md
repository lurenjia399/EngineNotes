// DPIScale是为了屏幕适配的，umg里的标准是1920*1080，也就是在umg里ui控件位置，大小在最终显示上都要乘DPIScale，所以这里除以DPIScale是为了得到umg下标准的位置。
void UUI_LockedByIndicator::UpdateTransform(...)
{
	TRACE_CPUPROFILER_EVENT_SCOPE(UI_LockedByIndicator_NativeTick_UpdateTransform);
	if (ArrowRootImage)
	{
		FWidgetTransform Transform;
		Transform.Scale = FVector2D(InRenderScale);
		Transform.Translation = InPosition / InViewportScale;
		Transform.Angle = bShouldRotateRoot ? InArrowAngle : ArrowRootImage->GetRenderTransformAngle();
		ArrowRootImage->SetRenderTransform(Transform);
	}
	if (ArrowHeadImage)
	{
		FWidgetTransform Transform;
		Transform.Scale = FVector2D(InRenderScale);
		Transform.Translation = InPosition / InViewportScale + InHeadOffset;
		Transform.Angle = bShouldRotateHead ? InArrowAngle : ArrowHeadImage->GetRenderTransformAngle();
		ArrowHeadImage->SetRenderTransform(Transform);
	}
}
