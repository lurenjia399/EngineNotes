```cpp
{
	
	GetEvaluateGraphExposedInputs().Execute(Context);
	
	// BlendPose 数组里面存的是链接的两个pose
	const int32 NumPoses = BlendPose.Num();
	// CurrentBlendTimes 数组里面存的是两个BlendTime
	const TArray<float>& CurrentBlendTimes = GetBlendTimes();
	bool bRequestedInertializationOnActiveChildIndexChange = false;
	
	if (NumPoses > 0)
	{
		// 混合的配置文件
		UBlendProfile* CurrentBlendProfile = GetBlendProfile();
		// 获取到当前应该选择哪个pose的索引，truePose就是0，falsePose就是1
		const int32 ChildIndex = GetActiveChildIndex();
		// 当前索引 ！= 之前索引。说明是在切换中
		if (ChildIndex != LastActiveChildIndex)
		{
			// 是否是第一次
			bool LastChildIndexIsInvalid = (LastActiveChildIndex == INDEX_NONE);
			// 计算WeightDifference，表示当前帧选择的pose，距离目标权重1还差多少。因为权重都是0到1，所以也可以看成还差多少比例
			const float CurrentWeight = PerBlendData[ChildIndex].Weight;
			const float DesiredWeight = 1.0f;
			const float WeightDifference = FMath::Clamp<float>(FMath::Abs<float>(DesiredWeight - CurrentWeight), 0.0f, 1.0f);
			// 计算剩余混合时间，还剩多少时间完全混合到当前帧选择的pose
			float RemainingBlendTime;
			if (LastChildIndexIsInvalid)
			{
				RemainingBlendTime = 0.0f;
			}
			else if (GetTransitionType() == EBlendListTransitionType::Inertialization)
			{
				// 这边是惯性混合部分，先省略
			}
			else
			{
				RemainingBlendTime = CurrentBlendTimes[ChildIndex] * WeightDifference;
			}
			for (int32 i = 0; i < PerBlendData.Num(); ++i)
			{
				PerBlendData[i].RemainingTime = RemainingBlendTime;
			}
			// 如果我们剩余混合时间为0，并且有上一个pose。就将上一个pose直接按权重为0混合，也就是去除掉上一个pose的影响
			if(RemainingBlendTime == 0.0f && !LastChildIndexIsInvalid)
			{
				BlendPose[LastActiveChildIndex].Update(
				Context.FractionalWeight(0.0f));
			}
			// 更新当前帧的混合数据
			for(int32 i = 0; i < PerBlendData.Num(); ++i)
			{
				FAlphaBlend& Blend = PerBlendData[i].Blend;
				// 设置剩余的混合时间
				Blend.SetBlendTime(RemainingBlendTime);
				// 设置权重的起始值和目标值
				if(i == ChildIndex)
				{
					Blend.SetValueRange(PerBlendData[i].Weight, 1.0f);

					if (CurrentBlendProfile)
					{
						Blend.ResetAlpha();
						PerBlendData[i].StartAlpha = Blend.GetAlpha();
					}
				}
				else
				{
					Blend.SetValueRange(PerBlendData[i].Weight, 0.0f);
				}
				if (CurrentBlendProfile)
				{
					PerBlendData[i].StartAlpha = Blend.GetAlpha();
				}
			}

			// 如果勾选了ResetChildOnActivation，并且当前权重为0。也就是第一次切换pose，会重新走遍初始化的流程
			if (GetResetChildOnActivation() && CurrentWeight <= ZERO_ANIMWEIGHT_THRESH)
			{
				FAnimationInitializeContext ReinitializeContext(Context.AnimInstanceProxy, Context.SharedContext);
				// reinitialize
				BlendPose[ChildIndex].Initialize(ReinitializeContext);
			}
			// 记录前一个pose索引
			LastActiveChildIndex = ChildIndex;
		}
		// 计算两个pose的权重和，然后归一化权重，放置权重和大于1吧
		float SumWeight = 0.0f;
		for (int32 i = 0; i < PerBlendData.Num(); ++i)
		{
			float& BlendWeight = PerBlendData[i].Weight;
			FAlphaBlend& Blend = PerBlendData[i].Blend;
			Blend.Update(Context.GetDeltaTime());
			BlendWeight = Blend.GetBlendedValue();
			SumWeight += BlendWeight;
		}
		if ((SumWeight > ZERO_ANIMWEIGHT_THRESH) && (FMath::Abs<float>(SumWeight - 1.0f) > ZERO_ANIMWEIGHT_THRESH))
		{
			float ReciprocalSum = 1.0f / SumWeight;
			for (int32 i = 0; i < PerBlendData.Num(); ++i)
			{
				PerBlendData[i].Weight *= ReciprocalSum;
			}
		}
		// 开始UpdatePose了
		for (int32 i = 0; i < BlendPose.Num(); ++i)
		{
			// 权重>0的才会走进来
			const float BlendWeight = PerBlendData[i].Weight;
			if (BlendWeight > ZERO_ANIMWEIGHT_THRESH)
			{
				// 叠加权重，递归网上找的时候会依次乘进去

					BlendWeight);
				// 更新方法，递归的网上执行了
				BlendPose[i].Update((i == ChildIndex) ? ChildContext : ChildContext.AsInactive());
			}
		}
	}
}
```

```cpp
float	AlphaLerp;

// 实际使用的已经完成的路程比例，将AlphaLerp曲线采样得出
float	AlphaBlend;

// 重置已经完成的路程比例，自然用已经完成的路程BlendedValue来计算出已经完成路程比例
void FAlphaBlend::ResetAlpha()
{
	float SmallerValue = FMath::Min(BeginValue, DesiredValue);
	float BiggerValue = FMath::Max(BeginValue, DesiredValue);
	float NewBlendedValue = FMath::Clamp(BlendedValue, SmallerValue, BiggerValue);
	if (BeginValue == DesiredValue)
	{
		SetAlpha(1.f);
	}
	else
	{
		AlphaLerp = (BlendedValue - BeginValue)/(DesiredValue - BeginValue);
		SetAlpha(AlphaLerp);
	}
	bNeedsToResetAlpha = false;
}
// 重置完成剩余路程所需的时间，用完成所有路程所需的时间 * 剩余的路程比例
void FAlphaBlend::ResetBlendTime()
{
	if(BlendTime <= 0.f)
	{
		BlendTimeRemaining = 0.f;
		SetAlpha(1.f);
	}
	else
	{
		BlendTimeRemaining = BlendTime * FMath::Abs(1.f - AlphaLerp);
	}
	bNeedsToResetBlendTime = false;
}

float FAlphaBlend::Update(float InDeltaTime)
{
	// 重置已经完成的路程比例，每次设置路程起点终点的时候就需要重置比例了
	if (bNeedsToResetAlpha)
	{
		ResetAlpha();
	}
	// 重置完成剩余路程所需的时间，每次设置 完成所有路程所需的时间就需要重置了
	if (bNeedsToResetBlendTime)
	{
		ResetBlendTime();
	}
	// 是否完成 = （已经完成的路程 == 路程终点）
	if( !IsComplete() )
	{
		// 完成剩余路程所需的时间 > 帧长
		if( BlendTimeRemaining > InDeltaTime )
		{
			// 当前帧完成的路径比例 = （未完成路程比例 / 完成剩余路程所需的时间）* 帧长
			const float BlendDelta = 1.f - AlphaLerp;
			//  已经完成的路程比例 = 已经完成的路程比例 + 当前帧完成的路径比例
			AlphaLerp += (BlendDelta / BlendTimeRemaining) * InDeltaTime;
			// 完成剩余路程所需的时间 = 完成剩余路程所需的时间 - 帧长
			BlendTimeRemaining -= InDeltaTime;
			SetAlpha(AlphaLerp);
		}
		// 完成剩余路程所需的时间 <= 帧长
		else
		{
			// 跑完所有路程还多余的时间
			float Overshoot = InDeltaTime - BlendTimeRemaining;
			// 完成剩余路程所需的时间 = 0.f
			BlendTimeRemaining = 0.f; 
			// 已经完成的路程比例 = 1.f
			SetAlpha(1.f);
			return Overshoot;
		}
	}
	return 0.f;
}
```

## TwoBoneIK和BasicIk
1 首先我们看下TwoBoneIK。节点分成两部分，第一部分是解算出JointBone的Location，第二部分是解算出JointBone的Rotation，从而得到JointBone的Transform。
第一部分的方式：我们已知RootBone，EndBone求JointBone的位置，通过三角形的三条边长能求出JointBone的位置，但是在三维空间中JointBone的位置是个圆环（圆环的中心轴是RootBone和EndBone两点连线），这时就需要JointTarget位置来确定圆环上的哪个点，通过RootBone，EndBone，JointTarget三点的平面和圆环相交两个交点，最终计算结果取和JointTarget在圆环中心轴同一侧的交点。
第二部分的方式：在求解出JointBone的Location后，我们还要求JointBone的Rotation和RootBone的Rotation。RootBone的Rotation计算方式就是通过RootBone到oldJointBone的向量 和 RootBone到newJointBone的向量 之间的旋转差（就是A向量旋转多少能到B向量）。JointBone的Rotation计算方式也是一样，通过oldJointBone到oldEndBone的向量 和 newJointBone到newEndBone的向量 之间的旋转差。至此RootBone的Transform和JointBone的Transform以及EndBone的Transform都求出。
```cpp
void SolveTwoBoneIK(...)
{
	FVector OutJointPos, OutEndPos;

	FVector RootPos = InOutRootTransform.GetLocation();
	FVector JointPos = InOutJointTransform.GetLocation();
	FVector EndPos = InOutEndTransform.GetLocation();

	// IK solver,第一部分
	AnimationCore::SolveTwoBoneIK(RootPos, JointPos, EndPos, JointTarget, Effector, OutJointPos, OutEndPos, UpperLimbLength, LowerLimbLength, bAllowStretching, StartStretchRatio, MaxStretchScale);

	// 计算RootBone的Transform
	{
		FVector const OldDir = (JointPos - RootPos).GetSafeNormal();
		FVector const NewDir = (OutJointPos - RootPos).GetSafeNormal();
		FQuat const DeltaRotation = FQuat::FindBetweenNormals(OldDir, NewDir);
		InOutRootTransform.SetRotation(DeltaRotation * InOutRootTransform.GetRotation());
		InOutRootTransform.SetTranslation(RootPos);
	}
	// 计算JointBone的Transform
	{
		FVector const OldDir = (EndPos - JointPos).GetSafeNormal();
		FVector const NewDir = (OutEndPos - OutJointPos).GetSafeNormal();
		FQuat const DeltaRotation = FQuat::FindBetweenNormals(OldDir, NewDir);
		InOutJointTransform.SetRotation(DeltaRotation * InOutJointTransform.GetRotation());
		InOutJointTransform.SetTranslation(OutJointPos);
	}
	// EndBone的Transform我们都是已知的
	InOutEndTransform.SetTranslation(OutEndPos);
}
```
2 ControlRig种的BasicIK，和TwoBoneIK一样的功能，第一部分走的接口都是一样的，区别在于第二部分，具体可查看 https://zhuanlan.zhihu.com/p/639376687 。
第二部分的方式：计算RootBone的Rotation是通过主轴，副轴来实现。拿手肘举例，手肘的主轴就是肘部到手腕的骨头，副轴就是手心朝向的方向。通过给主轴添加旋转我们可以将手放置在任何可以到达的地方，通过给副轴添加旋转，我们可以改变手心的朝向。

``` cpp
void FControlRigMathLibrary::SolveBasicTwoBoneIK(...)
{
	FVector RootPos = BoneA.GetLocation();
	FVector ElbowPos = BoneB.GetLocation();
	FVector EffectorPos = Effector.GetLocation();
	// 第一部分
	AnimationCore::SolveTwoBoneIK(RootPos, ElbowPos, EffectorPos, PoleVector, EffectorPos, ElbowPos, EffectorPos, BoneALength, BoneBLength, bEnableStretch, StretchStartRatio, StretchMaxRatio);
	BoneB.SetLocation(ElbowPos);
	Effector.SetLocation(EffectorPos);
	// 第二部分，计算Rotation，首先处理RootBone，JointBonde的处理是一样的
	// 把主轴用RootBone空间下表示
	FVector Axis = BoneA.TransformVectorNoScale(PrimaryAxis);
	// RootBone指向JointBone的向量
	FVector Target1 = BoneB.GetLocation() - BoneA.GetLocation();
	// JointBone的Location
	FVector BoneBLocation = BoneB.GetLocation();
	if (!Target1.IsNearlyZero() && !Axis.IsNearlyZero())
	{
		// RootBone指向JointBone的单位向量
		Target1 = Target1.GetSafeNormal();
		{
			// RootBone指向EndBone的单位向量
			FVector BaseDirection = (EffectorPos - RootPos).GetSafeNormal();
			// RootBone到JointBone和RootBone到EndBone夹角为0或者为180
			if (FMath::Abs(FVector::DotProduct(BaseDirection, Target1) - 1.0f) < KINDA_SMALL_NUMBER)
			{
				// 在JointBone的位置上添加一个JointBone指向JointTarget的方向的偏移
				BoneBLocation += (PoleVector - BoneBLocation).GetSafeNormal() * KINDA_SMALL_NUMBER*2.f;
				// RootBone指向JointBone上的单位向量
				Target1 = (BoneBLocation - BoneA.GetLocation()).GetSafeNormal();
			}
		}
		// RootBone空间下的主轴旋转到RootBone指向JointBone向量，所需的旋转
		FQuat Rotation1 = FQuat::FindBetweenNormals(Axis, Target1);
		// RootBone 先旋转（RootBone空间下的主轴旋转到RootBone指向JointBone向量所需的旋转），在旋转自己本身的旋转
		BoneA.SetRotation((Rotation1 * BoneA.GetRotation()).GetNormalized());
		// 副轴在RootBone空间下表示
		Axis = BoneA.TransformVectorNoScale(SecondaryAxis);
		// 填写了副轴的权重
		if (SecondaryAxisWeight > SMALL_NUMBER)
		{
			// RootBone到EndBone中心点 指向 JointBone的向量
			FVector Target2 = BoneBLocation - (Effector.GetLocation() + BoneA.GetLocation()) * 0.5f;
			if (!Target2.IsNearlyZero() && !Axis.IsNearlyZero())
			{
				Target2 = Target2 - FVector::DotProduct(Target2, Target1) * Target1;
				Target2 = Target2.GetSafeNormal();
				// 找到副轴旋转到Target2方向所需的旋转
				FQuat Rotation2 = FQuat::FindBetweenNormals(Axis, Target2);
				// RootBone 添加副轴的旋转
				BoneA.SetRotation((Rotation2 * BoneA.GetRotation()).GetNormalized());
			}
		}
	}
}
```

```cpp
{
	FTracker& Tracker = NodeTrackers.FindOrAdd(&InNode);
	
	if(!Tracker.Counter.HasEverBeenUpdated() || !Tracker.Counter.WasSynchronizedCounter(InContext.AnimInstanceProxy->GetUpdateCounter()))
	{
		Tracker.Status.CurrentWeight = 0.0f;
	}

	{
```

1 我们判断起跳的条件时Velocity的z轴大于0，或者是Character->JumpCurrentCount这个为1，通过后会进入JumpStart状态机。如果Velocity的z轴小于-10，或者是UCharacterMovementComponent::IsFalling这个方法，通过后会进入JumpLoop节点。这个JumpStart节点里根据水平速度的值，选择原地起跳动画序列还是移动起跳动画序列。

4 再JumpLoop节点上，如果离地高度小于100，怎会进入PreLand节点。离地高度就是每帧从root点向下打一条射线来计算。
