void FAnimNode_BlendListBase::Update_AnyThread(const FAnimationUpdateContext& Context)
{
	
	GetEvaluateGraphExposedInputs().Execute(Context);
	
	// BlendPose 数组里面存的是链接的两个pose
	const int32 NumPoses = BlendPose.Num();
	// CurrentBlendTimes 数组里面存的是两个BlendTime
	const TArray<float>& CurrentBlendTimes = GetBlendTimes();
	bool bRequestedInertializationOnActiveChildIndexChange = false;
	
	if (NumPoses > 0)
	{
		// 混合的配置文件
		UBlendProfile* CurrentBlendProfile = GetBlendProfile();
		// 获取到当前应该选择哪个pose的索引，truePose就是0，falsePose就是1
		const int32 ChildIndex = GetActiveChildIndex();
		// 当前索引 ！= 之前索引。说明是在切换中
		if (ChildIndex != LastActiveChildIndex)
		{
			// 是否是第一次
			bool LastChildIndexIsInvalid = (LastActiveChildIndex == INDEX_NONE);
			// 计算WeightDifference，表示当前帧选择的pose，距离目标权重1还差多少。因为权重都是0到1，所以也可以看成还差多少比例
			const float CurrentWeight = PerBlendData[ChildIndex].Weight;
			const float DesiredWeight = 1.0f;
			const float WeightDifference = FMath::Clamp<float>(FMath::Abs<float>(DesiredWeight - CurrentWeight), 0.0f, 1.0f);
			// 计算剩余混合时间，还剩多少时间完全混合到当前帧选择的pose
			float RemainingBlendTime;
			if (LastChildIndexIsInvalid)
			{
				RemainingBlendTime = 0.0f;
			}
			else if (GetTransitionType() == EBlendListTransitionType::Inertialization)
			{
				// 这边是惯性混合部分，先省略
			}
			else
			{
				RemainingBlendTime = CurrentBlendTimes[ChildIndex] * WeightDifference;
			}
			for (int32 i = 0; i < PerBlendData.Num(); ++i)
			{
				PerBlendData[i].RemainingTime = RemainingBlendTime;
			}
			// 如果我们剩余混合时间为0，并且有上一个pose。就将上一个pose直接按权重为0混合，也就是去除掉上一个pose的影响
			if(RemainingBlendTime == 0.0f && !LastChildIndexIsInvalid)
			{
				BlendPose[LastActiveChildIndex].Update(
				Context.FractionalWeight(0.0f));
			}
			// 更新当前帧的混合数据
			for(int32 i = 0; i < PerBlendData.Num(); ++i)
			{
				FAlphaBlend& Blend = PerBlendData[i].Blend;
				// 设置剩余的混合时间
				Blend.SetBlendTime(RemainingBlendTime);
				// 设置权重的起始值和目标值
				if(i == ChildIndex)
				{
					Blend.SetValueRange(PerBlendData[i].Weight, 1.0f);

					if (CurrentBlendProfile)
					{
						Blend.ResetAlpha();
						PerBlendData[i].StartAlpha = Blend.GetAlpha();
					}
				}
				else
				{
					Blend.SetValueRange(PerBlendData[i].Weight, 0.0f);
				}
				if (CurrentBlendProfile)
				{
					PerBlendData[i].StartAlpha = Blend.GetAlpha();
				}
			}

			// 如果勾选了ResetChildOnActivation，并且当前权重为0。也就是第一次切换pose，会重新走遍初始化的流程
			if (GetResetChildOnActivation() && CurrentWeight <= ZERO_ANIMWEIGHT_THRESH)
			{
				FAnimationInitializeContext ReinitializeContext(Context.AnimInstanceProxy, Context.SharedContext);
				// reinitialize
				BlendPose[ChildIndex].Initialize(ReinitializeContext);
			}
			// 记录前一个pose索引
			LastActiveChildIndex = ChildIndex;
		}
		// 计算两个pose的权重和，然后归一化权重，放置权重和大于1吧
		float SumWeight = 0.0f;
		for (int32 i = 0; i < PerBlendData.Num(); ++i)
		{
			float& BlendWeight = PerBlendData[i].Weight;
			FAlphaBlend& Blend = PerBlendData[i].Blend;
			Blend.Update(Context.GetDeltaTime());
			BlendWeight = Blend.GetBlendedValue();
			SumWeight += BlendWeight;
		}
		if ((SumWeight > ZERO_ANIMWEIGHT_THRESH) && (FMath::Abs<float>(SumWeight - 1.0f) > ZERO_ANIMWEIGHT_THRESH))
		{
			float ReciprocalSum = 1.0f / SumWeight;
			for (int32 i = 0; i < PerBlendData.Num(); ++i)
			{
				PerBlendData[i].Weight *= ReciprocalSum;
			}
		}
		// 开始UpdatePose了
		for (int32 i = 0; i < BlendPose.Num(); ++i)
		{
			// 权重>0的才会走进来
			const float BlendWeight = PerBlendData[i].Weight;
			if (BlendWeight > ZERO_ANIMWEIGHT_THRESH)
			{
				// 叠加权重，递归网上找的时候会依次乘进去
				FAnimationUpdateContext ChildContext = Context.FractionalWeight(
					BlendWeight);
				// 更新方法，递归的网上执行了
				BlendPose[i].Update((i == ChildIndex) ? ChildContext : ChildContext.AsInactive());
			}
		}
	}
}

// 路程表示pose从开始切换到切换完成
// 完成所有路程所需的时间
UPROPERTY(EditAnywhere, Category = "Blend")  
float	BlendTime;

// 完成剩余路程所需的时间
float	BlendTimeRemaining;

// 已经完成的路程比例。由BlendTimeRemaining和帧长计算得出
float	AlphaLerp;

// 实际使用的已经完成的路程比例，将AlphaLerp曲线采样得出
float	AlphaBlend;

// 已经完成的路程，由AlphaBlend比例插值计算出
float BlendedValue;

// 路程起点
float BeginValue;

// 路程终点
float DesiredValue;

// 下面列举一些关键方法：

// 重置已经完成的路程比例，自然用已经完成的路程BlendedValue来计算出已经完成路程比例
void FAlphaBlend::ResetAlpha()
{
	float SmallerValue = FMath::Min(BeginValue, DesiredValue);
	float BiggerValue = FMath::Max(BeginValue, DesiredValue);
	float NewBlendedValue = FMath::Clamp(BlendedValue, SmallerValue, BiggerValue);
	if (BeginValue == DesiredValue)
	{
		SetAlpha(1.f);
	}
	else
	{
		AlphaLerp = (BlendedValue - BeginValue)/(DesiredValue - BeginValue);
		SetAlpha(AlphaLerp);
	}
	bNeedsToResetAlpha = false;
}
// 重置完成剩余路程所需的时间，用完成所有路程所需的时间 * 剩余的路程比例
void FAlphaBlend::ResetBlendTime()
{
	if(BlendTime <= 0.f)
	{
		BlendTimeRemaining = 0.f;
		SetAlpha(1.f);
	}
	else
	{
		BlendTimeRemaining = BlendTime * FMath::Abs(1.f - AlphaLerp);
	}
	bNeedsToResetBlendTime = false;
}

float FAlphaBlend::Update(float InDeltaTime)
{
	// 重置已经完成的路程比例，每次设置路程起点终点的时候就需要重置比例了
	if (bNeedsToResetAlpha)
	{
		ResetAlpha();
	}
	// 重置完成剩余路程所需的时间，每次设置 完成所有路程所需的时间就需要重置了
	if (bNeedsToResetBlendTime)
	{
		ResetBlendTime();
	}
	// 是否完成 = （已经完成的路程 == 路程终点）
	if( !IsComplete() )
	{
		// 完成剩余路程所需的时间 > 帧长
		if( BlendTimeRemaining > InDeltaTime )
		{
			// 当前帧完成的路径比例 = （未完成路程比例 / 完成剩余路程所需的时间）* 帧长
			const float BlendDelta = 1.f - AlphaLerp;
			//  已经完成的路程比例 = 已经完成的路程比例 + 当前帧完成的路径比例
			AlphaLerp += (BlendDelta / BlendTimeRemaining) * InDeltaTime;
			// 完成剩余路程所需的时间 = 完成剩余路程所需的时间 - 帧长
			BlendTimeRemaining -= InDeltaTime;
			SetAlpha(AlphaLerp);
		}
		// 完成剩余路程所需的时间 <= 帧长
		else
		{
			// 跑完所有路程还多余的时间
			float Overshoot = InDeltaTime - BlendTimeRemaining;
			// 完成剩余路程所需的时间 = 0.f
			BlendTimeRemaining = 0.f; 
			// 已经完成的路程比例 = 1.f
			SetAlpha(1.f);
			return Overshoot;
		}
	}
	return 0.f;
}

FAnimNodeRelevancyStatus FAnimSubsystemInstance_NodeRelevancy::UpdateNodeRelevancy(...)
{
	FTracker& Tracker = NodeTrackers.FindOrAdd(&InNode);
	
	if(!Tracker.Counter.HasEverBeenUpdated() || !Tracker.Counter.WasSynchronizedCounter(InContext.AnimInstanceProxy->GetUpdateCounter()))
	{
		Tracker.Status.CurrentWeight = 0.0f;
	}

	if(!Tracker.Counter.IsSynchronized_All(InContext.AnimInstanceProxy->GetUpdateCounter()))
	{
		Tracker.Counter.SynchronizeWith(InContext.AnimInstanceProxy->GetUpdateCounter());
		Tracker.Status.PreviousWeight = Tracker.Status.CurrentWeight;
		Tracker.Status.CurrentWeight = InContext.GetFinalBlendWeight();
	}
	
	return Tracker.Status;
}

### StrideBlend 步幅计算

![image.png](https://gitee.com/lurenjia399/image/raw/master/image/202412181125374.png)

1 首先获取我们的速度。
2 我们需要有两根不同步态下的步幅曲线，也就是StrideBlend_N_Walk和StrideBlend_N_Run，横坐标都是速度，纵坐标是步幅（范围是从0到1的）。

![image.png](https://gitee.com/lurenjia399/image/raw/master/image/202412251648197.png)

![image.png](https://gitee.com/lurenjia399/image/raw/master/image/202412271420659.png)
