### 1 UPendingNetGame::InitNetDriver

```cpp
void UPendingNetGame::InitNetDriver()
{
	LLM_SCOPE(ELLMTag::Networking);

	if (!GDisallowNetworkTravel)
	{
		// 开始这和服务器创建一样，都是创建出NetDriver
		if (GEngine->CreateNamedNetDriver(this, NAME_PendingNetDriver, NAME_GameNetDriver))
		{
			NetDriver = GEngine->FindNamedNetDriver(this, NAME_PendingNetDriver);
		}
		// 然后就是初始化链接
		if( NetDriver->InitConnect( this, URL, ConnectionError ) )
		{
			UNetConnection* ServerConn = NetDriver->ServerConnection;

			FNetDelegates::OnPendingNetGameConnectionCreated.Broadcast(this);

			// Kick off the connection handshake
			// 触发握手流程
			if (ServerConn->Handler.IsValid())
			{
				ServerConn->Handler->BeginHandshaking(
					FPacketHandlerHandshakeComplete::CreateUObject(this, &UPendingNetGame::SendInitialJoin));
			}
			else
			{
				SendInitialJoin();
			}
		}
		else
		{
			UE_LOG(LogNet, Warning, TEXT("error initializing the network stack"));
			GEngine->DestroyNamedNetDriver(this, NetDriver->NetDriverName);
			NetDriver = NULL;
		}
	}
}
```
不是很复杂，首先就是通过CreateNamedNetDriver创建出NetDriver，这部分和服务器的UWorld::Listen方法一样，服务器之后会调用NetDriver的InitListen方法，而这里客户端会调用NetDriver的InitConnect方法。

```cpp
void UWorld::NotifyControlMessage(...)
{
	switch (MessageType)
	{
		case NMT_Hello:
		{
			uint8 IsLittleEndian = 0;
			uint32 RemoteNetworkVersion = 0;
			uint32 LocalNetworkVersion = FNetworkVersion::GetLocalNetworkVersion();
			FString EncryptionToken;

			if (FNetControlMessage<NMT_Hello>::Receive(Bunch, IsLittleEndian, RemoteNetworkVersion, EncryptionToken))
			{
				if (!FNetworkVersion::IsNetworkCompatible(LocalNetworkVersion, RemoteNetworkVersion))
				{
					// 信息不匹配，就会发送NMT_Upgrade包
					FNetControlMessage<NMT_Upgrade>::Send(Connection, LocalNetworkVersion);
					Connection->FlushNet(true);
					Connection->Close();
				}
				else
				{
					if (EncryptionToken.IsEmpty())
					{
						// 发送Challenge包
						Connection->SendChallengeControlMessage();
					}
				}
			}
			break;
		}
	}
}
```

```cpp
			// Server accepted connection.

				// 并设置标志位,客户端可以加载地图了

			{
				URL.Map.Empty();
			}
			break;
		}
	}
}
```

```cpp
APlayerController* UWorld::SpawnPlayActor(...)
{
	if (AGameModeBase* const GameMode = GetAuthGameMode())
	{
		// Give the GameMode a chance to accept the login
		APlayerController* const NewPlayerController = GameMode->Login(NewPlayer, RemoteRole, *InURL.Portal, Options, UniqueId, Error);
```

# 2 Actor同步
NetDriver是驱动网络相关的操作，同步也是由NetDriver::tick执行的
```cpp
void UWorld::Tick(...)
{
	{
		// 这里客户端和服务器都会调用到TickDispatch
		BroadcastTickDispatch(DeltaSeconds);
		BroadcastPostTickDispatch();
		if( NetDriver && NetDriver->ServerConnection )
		{
			TickNetClient( DeltaSeconds );
		}
	}
	
	// 客户端TickFlush
	if (!IsRunningDedicatedServer())
		{
			// Update net and flush networking.
			// Tick all net drivers
			{
				BroadcastTickFlush(RealDeltaSeconds); // note: undilated time is being used here
			}
	
			// PostTick all net drivers
			{
				LLM_SCOPE(ELLMTag::Networking);
				BroadcastPostTickFlush(RealDeltaSeconds); // note: undilated time is being used here
			}
		}
	}
	// 这里会执行Actor的Tick，也就是RunTickGroup
	// 服务器TickFlush
	if (IsRunningDedicatedServer())
	{
		// Update net and flush networking.
		// Tick all net drivers
		{
			SCOPE_CYCLE_COUNTER(STAT_NetBroadcastTickTime);
			BroadcastTickFlush(RealDeltaSeconds); // note: undilated time is being used here
		}
	}
	
```
客户端和服务器都会优先执行TickDispatch，然后客户端执行TickFlush，进而执行RunTickGroup，最后服务器执行TickFlush
## 1 Server UNetDriver::TickFlush
```cpp
void UNetDriver::TickFlush(float DeltaSeconds)
{
	if (IsServer() && ClientConnections.Num() > 0 && !bSkipServerReplicateActors)
	{
		// Update all clients.
		int32 Updated = ServerReplicateActors( DeltaSeconds );
		// 客户端走
		if( ServerConnection )
		{
			// Queue client voice packets in the server's voice channel
			ProcessLocalClientPackets();
			// 客户端这会执行NetConnection的Tick
			// 在NetConnection的Tick里面会执行ChannelsTick数组的Tick，也就是会执行到ProcessQueuedBunches方法
			ServerConnection->Tick(DeltaSeconds);
		}
		// 服务器走
		else
		{
			// Queue up any voice packets the server has locally
			ProcessLocalServerPackets();
		}
	}
}
```
### Server UNetDriver::ServerReplicateActors
```cpp
int32 UNetDriver::ServerReplicateActors(float DeltaSeconds)
{
	// 这个方法的返回值代表了有几个ClientConnection可以Tick，也就是说有几个客户端需要考虑同步
	// 并切赋值了Connection中的ViewTarget
	{
		const int32 NumClientsToTick = ServerReplicateActors_PrepConnections( DeltaSeconds );  
		if ( NumClientsToTick == 0 )  
		{  
		    return 0;  
		}
	}
	// 构建需要考虑复制的Actor
	{
		TArray<FNetworkObjectInfo*> ConsiderList;
		ConsiderList.Reserve( GetNetworkObjectList().GetActiveObjects().Num() );
		ServerReplicateActors_BuildConsiderList( ConsiderList, ServerTickTime );
	}
	// 根据ViewTarget的位置，确定PriorityList，PriorityActors两个数组，带有优先级的Actor，这里面的Actor应该是ConsiderList的子集
	// 然后是对排序后的Actor进行处理，没有ActorChannel会创建，然后同步数据到客户端
	{
		FActorPriority* PriorityList    = NULL;  
		FActorPriority** PriorityActors = NULL;
		for ( int32 i=0; i < ClientConnections.Num(); i++ )  
		{
			if (Connection->ViewTarget)  
			{
				// 这里筛选排序
				const int32 FinalSortedCount = ServerReplicateActors_PrioritizeActors( Connection, ConnectionViewers, ConsiderList, bCPUSaturated, PriorityList, PriorityActors );
				const int32 LastProcessedActor = ServerReplicateActors_ProcessPrioritizedActors( Connection, ConnectionViewers, PriorityActors, FinalSortedCount, Updated );
				
			}
		}
	}
	// 在处理Actor的时候，因为连接饱和了导致没处理完，剩下还未处理的在下一帧处理
	{
		for ( int32 k=LastProcessedActor; k<FinalSortedCount; k++ )
		{
			// 根据不同情况设置bPendingNetUpdate标志位，下一帧同步这个Actor
			PriorityActors[k]->ActorInfo->bPendingNetUpdate = true;
		}
	}
	// 记录Connection这次同步帧号
	Connection->LastProcessedFrame = ReplicationFrame;
	// 将这次处理的Connection放到ClientConnection数组的后边，以便下一帧能优先处理
	{
		if (NumClientsToTick < ClientConnections.Num())
		{
			int32 NumConnectionsToMove = NumClientsToTick;
			while (NumConnectionsToMove > 0)
			{
				UNetConnection *Connection = ClientConnections[0];
				ClientConnections.RemoveAt(0,1);
				ClientConnections.Add(Connection);
				NumConnectionsToMove--;
			}
		}
	}
	// 关闭掉之前记录的需要关闭的Connection
	{
		for (UNetConnection* ConnectionToClose : ConnectionsToClose)
		{
			ConnectionToClose->Close();
		}
	}
}
```

```cpp
void UNetDriver::ServerReplicateActors_BuildConsiderList(...)
{
	// 遍历ActiveNetworkObjects数组，这个数组会在AActor::SetReplicates这个方法中通过AddNetworkActor方法被添加
	// 也就是所有被设置成可复制的Actor
	for ( const TSharedPtr<FNetworkObjectInfo>& ObjectInfo : GetNetworkObjectList().GetActiveObjects() )
	{
		FNetworkObjectInfo* ActorInfo = ObjectInfo.Get();
		// 如果Actor不在同步过程中 并且 同步时间没到 不同步跳过
		if ( !ActorInfo->bPendingNetUpdate && World->TimeSeconds <= ActorInfo->NextUpdateTime )
		{
			continue;
		}
		AActor* Actor = ActorInfo->Actor;
		// Actor应该被销毁，不同步跳过
		if ( Actor->IsPendingKillPending() )
		{
			ActorsToRemove.Add( Actor );
			continue;
		}
		// Actor没有Role，不同步跳过
		if ( Actor->GetRemoteRole() == ROLE_None )  
		{  
			ActorsToRemove.Add( Actor );  
			continue;  
		}
		// Actor不属于当前这个NetDriver，不同步跳过
		if (Actor->GetNetDriverName() != NetDriverName)  
		{
		    continue;  
		}
		// Actor还没有初始化，不同步跳过
		if ( !Actor->IsActorInitialized() )  
		{  
		    continue;  
		}
		// 这个自己加的，Actor延迟BeginPlay，不同步跳过
		if ( Actor->bDeferredBeginPlay )  
		{  
		    continue;  
		}
		// Actor所处的Level正在切换过程中，不同步跳过
		ULevel* Level = Actor->GetLevel();  
		if ( Level->HasVisibilityChangeRequestPending() || Level->bIsAssociatingLevel )  
		{  
		    continue;  
		}
		// 是一个休眠的网络Actor，不同步跳过，不太清楚干嘛的
		if ( IsDormInitialStartupActor(Actor) )
		{
			ActorsToRemove.Add( Actor );
			continue;
		}
		// Actor下一次需要同步的时间 = 当前世界的时间 + 随机值 + 配置时间
		ActorInfo->NextUpdateTime = World->TimeSeconds + UpdateDelayRandomStream.FRand() * ServerTickTime + NextUpdateDelta;
		// 最终赋值结果数组
		OutConsiderList.Add( ActorInfo );
		// 调用一个赋值之前的方法，可以放些操作？
		Actor->CallPreReplication( this );
	}
}
```

#### 4 UNetDriver::ServerReplicateActors_ProcessPrioritizedActors
```cpp
int32 UNetDriver::ServerReplicateActors_ProcessPrioritizedActors(...)
{
	// 遍历我们排好序的Actor，拿到ActorInfo
	for ( int32 j = 0; j < FinalSortedCount; j++ )
	{
		FNetworkObjectInfo*	ActorInfo = PriorityActors[j]->ActorInfo;
		UActorChannel* Channel = PriorityActors[j]->Channel;
		// 当前处理的Actor的ActorChannel还没创建，或者是已经创建了
		if ( !Channel || Channel->Actor )
		{
			AActor* Actor = ActorInfo->Actor;
			// 表示了此次Actor同步，是否与Connection相关联
			bool bIsRelevant = false;
			// 当前Actor所在的Level，在对应Connection客户端是否加载了
			const bool bLevelInitializedForActor = IsLevelInitializedForActor( Actor, Connection );
			// 如果客户端加载了Actor所处的Level
			if ( bLevelInitializedForActor )
			{
				// GetTearOff 代表当前Actor不在参与网络复制，在客户端已经复制了的就改变其Role为ROLE_Authority
				// 不还没创建对应的ActorChannel或者是netDriver创建的持续时间 - Actor上一次与Connection相关联的时间 > 1。也就是没超过1s都检查下此Actor是否还与Connection相关联。
				if ( !Actor->GetTearOff() && ( !Channel || ElapsedTime - Channel->RelevantTime > 1.0 ) )
				{
					if ( IsActorRelevantToConnection( Actor, ConnectionViewers ) )
					{
						bIsRelevant = true;
					}
				}
			}
			// 这个标志位表示了，此Actor最近是否与Connection相关联
			// 如果此次同步Actor是相关联的 或者 当前距离上一次相关联时间小于5s 都是为true
			const bool bIsRecentlyRelevant = bIsRelevant || ( Channel && ElapsedTime - Channel->RelevantTime < RelevantTimeout ) || (ActorInfo->ForceRelevantFrame >= Connection->LastProcessedFrame);

			if ( bIsRecentlyRelevant )
			{
				FinalRelevantCount++;
				// 进行一系列判断(这里都省略掉了)
				// 判断是否能够给当前Actor创建ActorChannel
				if ( Channel == NULL)
				{
					// Actor所处的Level已经加载好了
					if ( bLevelInitializedForActor )
					{
						// 给当前Actor创建ActorChannel，并赋值ChannelActor
						Channel = (UActorChannel*)Connection->CreateChannelByName( NAME_Actor, EChannelCreateFlags::OpenedLocally );
						if ( Channel )
						{
							Channel->SetChannelActor(Actor, ESetChannelActorFlags::None);
						}
					}
					// Actor所处的Level已经没加载好，所以不能同步，也就不创建ActorChannel
					// 但如果Actor的同步更新频率很快，则我们就设置下Actor下次更新时间
					else if ( Actor->NetUpdateFrequency < 1.0f )
					{
						ActorInfo->NextUpdateTime = World->TimeSeconds + 0.2f * FMath::FRand();
					}
				}

				if ( Channel )
				{
					// 如果此次Actor是相关的，就记录一下相关时间
					if ( bIsRelevant )
					{
						Channel->RelevantTime = ElapsedTime + 0.5 * UpdateDelayRandomStream.FRand();
					}
					// Actor准备好了，可以发数据？
					if ( Channel->IsNetReady( 0 ) )
					{
						// 这个就是ActorChannel同步Actor数据的方法
						if ( Channel->ReplicateActor() )
						{
							ActorUpdatesThisConnectionSent++;
							// 这边记录了下Actor同步的时间，是world的绝对时间
							ActorInfo->LastNetReplicateTime = World->TimeSeconds;
						}
						ActorUpdatesThisConnection++;
						OutUpdated++;
					}
				}
			}
			// 如果最近不相关，并且ActorChannel还存在
			// 如果Actor所处的Level还不存在 或者 Actor不是一个可同步Actor（直接放在地图里的可同步Actor）那就close ActorChannel
			if ( ( !bIsRecentlyRelevant || Actor->GetTearOff() ) && Channel!=NULL)
			{
				if ( !bLevelInitializedForActor || !Actor->IsNetStartupActor() )
				{
					// 关闭ActorChannel
					Channel->Close(Actor->GetTearOff() ? EChannelCloseReason::TearOff : EChannelCloseReason::Relevancy);
				}
			}
		}
	}
	// 返回的是已经处理的Actor也就是同步的或者是提前continue的
	return FinalSortedCount;
}
```
这个方法就是对根据优先级排好序需要同步Actor的处理。
1 首先确定是否相关。如果距离上一次记录的相关时间超过1s，就调用IsActorRelevantToConnection方法（这个方法里会用到NetCullDistanceSquared变量，Actor距离NetConnection的距离需要小于这个变量）检查，并将赋值bIsRelevant，表示这次同步是否相关。
2 赋值bIsRecentlyRelevant变量。这个变量表示Actor这次或者是RelevantTimeout（引擎中配置5s）秒前是相关的。
3 创建ActorChannel。如果我们还没有创建ActorChannel就创建一个，并调用SetChannelActor初始化Channel。
4 记录相关时间。是NetDriver持续时间 + 0.5 乘上随机数。
5 同步数据。ActorChannel调用ReplicateActor方法。
6 如果Actor最近不相关（此次同步是不相关的，RelevantTimeout秒前也是不相关的），并且Channel还存在，就关闭Channel。
7 返回值。返回了处理了的Actor数量，如果正常返回值会等于排好序Actor的数量，如果同步中途Connection饱和了，这一次就停止同步，提前返回这一次处理了多少Actor。

##### 4 UPackageMapClient::InternalWriteObject
```cpp
void UPackageMapClient::InternalWriteObject( FArchive & Ar, FNetworkGUID NetGUID, UObject* Object, FString ObjectPathName, UObject* ObjectOuter )
{
	// 
	/*
		GuidCache->ShouldAsyncLoad() 客户端是否支持异步加载Object
		IsNetGUIDAuthority() 是否在服务器上
		!GuidCache->IsExportingNetGUIDBunch 不能是ExportingNetGUIDBunch的流程
		!bNoLoad 客户端可以加载此Object
	*/
	// 满足条件后，会将Object的NetGUID添加到MustBeMappedGuidsInLastBunch这个数组里
	if ( GuidCache->ShouldAsyncLoad() && IsNetGUIDAuthority() && !GuidCache->IsExportingNetGUIDBunch && !bNoLoad )
	{
		MustBeMappedGuidsInLastBunch.AddUnique( NetGUID );
	}
	// 把NetGUID写入
	Ar << NetGUID;
	if ( NetGUID.IsDefault() )
	{
		ExportFlags.bHasPath = 1;
		Ar << ExportFlags.Value;
	}
	// 上边的ExportNetGUID这个方法就会走这里，如果我们Object通过ShouldSendFullPath这个方法，他就会写入bHasPath，意思是静态的Object,或者说是网络名称稳定的那种
	else if ( GuidCache->IsExportingNetGUIDBunch )
	{
		if ( Object != NULL )
		{
			ExportFlags.bHasPath = ShouldSendFullPath( Object, NetGUID ) ? 1 : 0;
		}
		else
		{
			ExportFlags.bHasPath = ObjectPathName.IsEmpty() ? 0 : 1;
		}
		ExportFlags.bNoLoad	= bNoLoad ? 1 : 0;
```

### 2 UActorChannel::ReceivedBunch
```cpp
void UActorChannel::ReceivedBunch( FInBunch & Bunch )
{
	TArray<TPair<FNetworkGUID, UObject*>> QueuedObjectsToTrack;
	if (Connection->Driver->IsServer())
	{
		// 在服务器上如果还有MustBeMappedGUIDs,就说明有问题
		// bHasMustBeMappedGUIDs这个字段标识了，客户端收到同步Object，是否用异步的方式创建
		if (Bunch.bHasMustBeMappedGUIDs)
		{
			Bunch.SetError();
			return;
		}
	}
	else
	{
		if (Bunch.bHasMustBeMappedGUIDs)
		{
			// 从Bunch中拿出有多少个GUID的Object，可以被异步加载出来
			uint16 NumMustBeMappedGUIDs = 0;
			Bunch << NumMustBeMappedGUIDs;
			// 设置数组的大小
			QueuedObjectsToTrack.SetNum(NumMustBeMappedGUIDs);
			// 拿到GUIDCache
			FNetGUIDCache* GuidCache = Connection->Driver->GuidCache.Get();
			// 遍历
			for (int32 i = 0; i < NumMustBeMappedGUIDs; i++)
			{
				// 拿出GUID
				FNetworkGUID NetGUID;
				Bunch << NetGUID;
				// 如果不支持异步加载，就直接返回掉
				if (!GuidCache->ShouldAsyncLoad())
				{
					continue;
				}
				// 在Cache中找到了NetGUID对应的CacheObject
				if (FNetGuidCacheObject const * const GuidCacheObject = GuidCache->GetCacheObject(NetGUID))
				{
					// 如果Cache中的Object已经加载出来了，放到QueuedObjectsToTrack数组里
					if (UObject* const Object = GuidCacheObject->Object.Get())
					{
						QueuedObjectsToTrack.Emplace(NetGUID, Object);
					}
					else
					// Cache中的Object没有加载出来
					{
						// 放到PendingGuidResolves数组里
						PendingGuidResolves.Add(NetGUID);
						// 将此Channel添加到ChannelsToTick数组中，等待被Tick
						Connection->StartTickingChannel(this);
						// 客户端会把未解析出静态Object的NetGUID放到ActorChannel的QueuedBunchObjectReferences数组中，也会放到NetGUIDCache的QueuedBunchObjectReferences数组中
						QueuedBunchObjectReferences.Add(GuidCache->TrackQueuedBunchObjectReference(NetGUID, nullptr));
					}
				}
				// 如果在Cache中找不见NetGUID对应的CacheObject，就说明有问题
				// 这也说明了在ReceivedBunch之前，就对静态的Object完成了注册
				// 静态的Object都是在ExportBunch中是在ReceivedRawBunch方法里注册的
				else
				{
					Bunch.SetError();
					return;
				}
			}
		}
		if (Actor == NULL && Bunch.bOpen)
		{
			// 序列化出ActorNetGUID
			Bunch << ActorNetGUID;
		}
		if (PendingGuidResolves.Num() > 0 || QueuedBunches.Num() > 0 || Connection->KeepProcessingActorChannelBunchesMap.Contains(ActorNetGUID) ||
			 (Connection->Driver->ShouldQueueBunchesForActorGUID(ActorNetGUID)))
		{
			// 把当前这个Bunch加入QueueBunches里
			QueuedBunches.Add(new FInBunch(Bunch));
			Connection->StartTickingChannel(this);
			return;
		}
	}
	// 这个EnableProcessNewActorOnly是我们自己加的，ue5没有
	if (EnableProcessNewActorOnly)
	{
		// Azure SpawnedNewActor
		// 如果ActorChannel里面没有Actor，就根据Bunch里的数据创建一个新的
		if (Actor == nullptr)
		{
			SpawnedNewActor(Bunch);
		}

		// We can process this bunch now
		ProcessBunchWithoutNewActor(Bunch);

		// Azure PostSpawnedNewActor
		if (bSpawningNewActor)
		{
			PostSpawnedNewActor();
		}
	}
	else
	{
		// We can process this bunch now
		ProcessBunch(Bunch);
	}
}
```

2 每一帧会把所有的Actor都同步么？
不会。每一帧会都会构建需要同步的Actor数组并计算下一次需要同步的时间（这个时间是当前world时间 + 小范围的随机值 + Actor的同步频率的配置），每次在构建同步数组的时间会看是否到达了时间，如果到了才有可能加入数组中。每一个ClientConnection都会有可用的空间长度，每次同步了一个Actor之后会计算这个空间长度是否足够，不够了就说明饱和了，剩余还未同步的Actor就给下一帧同步。
3 在客户端spawn一个actor，并调用AActor::SetReplicates方法，服务器会有这个actor么？
不会，客户端创建的一定只在客户端，setreplicates只是会把actor放在GetNetworkObjectList()数组中，但是这是在客户端里面，服务器同步的话，必须是放在服务器的GetNetworkObjectList()数组中。
4 replicate怎么运作的？客户端服务器都走的代码，比如actor是怎么同步过来的？
看上去如果在服务器创建一个可复制的actor，他会自动同步到客户端。如果客户端在创建一个可复制的actor，那就会在客户端存在两个actor。
5 客户端创建一个replicate的actor，服务器会同步么？
不会欸，这样就只是在客户端创建了一个自己的actor，他的role是authority，remoterole是none

7 项目中，联机模式下客户端链接服务器之后，服务器创建playerController之后创建pawn，地图是加载完的。单机模式下创建playerController之后创建pawn的时候，地图没加载完？
看上去联机模式下，服务器在加载完地图等待客户端的链接，当客户端加载完地图发送消息给服务器后，服务器创建playerContrller进一步创建pawn，这时候地图肯定是加载完的。
但是在单机模式下，在loadmap中会首先通过localplayer创建对应的PlayerController进一步创建pawn，但是地图加载完的事件PostLoadMapCaller是在PlayerController创建之后广播，也就是会认为地图没有加载完。单机模式下会有localPlayer以及netDriver会有serverconnection。

```cpp
			continue;
		}

		if (UFunction* RepNotifyFunc = InObjectClass->FindFunctionByName(Parents[ParentIndex].Property->RepNotifyFunc))

			Parents[ParentIndex].RepNotifyNumParams = RepNotifyFunc->NumParms;
		}

	BuildShadowOffsets<ERepBuildType::Class>(InObjectClass, Parents, Cmds, ShadowDataBufferSize);
	// 最终设置变量
	Owner = InObjectClass;
```

```cpp
{
```

```cpp
	RepState->LastCompareIndex = RepChangelistState->CompareIndex;

	{
		const int32 HistoryIndex = i % FRepChangelistState::MAX_CHANGE_HISTORY;
		FRepChangedHistory& HistoryItem = RepChangelistState->ChangeHistory[HistoryIndex];
		TArray<uint16> Temp = MoveTemp(Changed);
		MergeChangeList(Data, HistoryItem.Changed, Temp, Changed);
	}

	RepState->LastChangelistIndex = RepChangelistState->HistoryEnd;
	// 此次有属性改变了，Changed里面有东西

	// 记录此时Writer的大小
	const int32 NumBits = Writer.GetNumBits();

	SendProperties(RepState, ChangeTracker, Data, ObjectClass, Writer, Changed, RepChangelistState->SharedSerialization);
```

```cpp
	第四部分：如果actor是第一次创建，就调用其PostNetInit方法。

void UActorChannel::ProcessBunch( FInBunch & Bunch )
{
	//第一部分
	if( Actor == NULL )
	{
		AActor* NewChannelActor = NULL;
		bSpawnedNewActor = Connection->PackageMap->SerializeNewActor(Bunch, this, NewChannelActor);
		SetChannelActor(NewChannelActor, Flags);
		NotifyActorChannelOpen(Actor, Bunch);
		RepFlags.bNetInitial = true;
	}
	//第二部分
	while ( !Bunch.AtEnd() && Connection != NULL && Connection->GetConnectionState() != USOCK_Closed )
	{
		// 这个会读取UObject，如果是Actor就直接返回这个ActorChannel里的Actor，如果是Object就会根据Bunch反序列化出来一个Object
		UObject* RepObj = ReadContentBlockPayload( Bunch, Reader, bHasRepLayout );
		// 创建Object的属性复制器，也就是再客户端创建属性同步需要的那些对象
		TSharedRef< FObjectReplicator > & Replicator = FindOrCreateReplicator( RepObj );
		// 在这个ObjectReplicator::RecivedBunch方法里面,如果客户端通过SerializeObject无法序列化出Object，这个标志位就为false
		bool bHasUnmapped = false;
		// 这个里面是客户端比较属性是否改变，改变了的会存到RepNotifies数组里面
		if ( !Replicator->ReceivedBunch( Reader, RepFlags, bHasRepLayout, bHasUnmapped ) )
		{
			// 有问题的情况才会走进来
			Connection->Close();
			return;
		}
		// 如果客户端没序列化出来，就会把Replicator添加到UnmappedReplicators数组里面
		if ( bHasUnmapped )
		{
			// 这个数组是会在TickFlush里面执行的
			Connection->Driver->UnmappedReplicators.Add( &Replicator.Get() );
		}
	}
	//第三部分
	for (auto RepComp = ReplicationMap.CreateIterator(); RepComp; ++RepComp)
	{
		TSharedRef<FObjectReplicator>& ObjectReplicator = RepComp.Value();
		// 这个方法里会调用CallRepNotifies方法，也就是客户端的onrep
		ObjectReplicator->PostReceivedBunch();
	}
	// 第四部分
	if (Actor && bSpawnedNewActor)
	{
		Actor->PostNetInit();
	}
}
```

```cpp
static bool ReceivePropertyHelper(...)
{
	// 拿到我们的属性
	const FRepLayoutCmd& Cmd = Cmds[CmdIndex];  
	const FRepParentCmd& Parent = Parents[Cmd.ParentIndex];
	// 如果 GuidReferencesMap 这个数组存在，开启TrackedGUID
	if (GuidReferencesMap)
	{
		Bunch.PackageMap->ResetTrackedGuids(true);
	}
	// 如果有OnRep方法
	if (RepNotifies != nullptr && INDEX_NONE != Parent.RepNotifyNumParams)
	{
		// Data里的数据是客户端存在的值，把这个值赋给RepState->StaticBuffer。也就是赋值给ShadowData
		StoreProperty(Cmd, ShadowData + Cmd, Data + SwappedCmd);
		// 序列化属性，也就是从Bunch里面拿出数据赋值给Data
		Cmd.Property->NetSerializeItem(Bunch, Bunch.PackageMap, Data + SwappedCmd);
		// 具体检测属性是否改变的方法，也就是比较Data和ShadowData数据是否一样。如果改变了并有Onrep方法，就添加到RepNotifies里面
		if (Parent.RepNotifyCondition == REPNOTIFY_Always || !PropertiesAreIdentical(Cmd, ShadowData + Cmd, Data + SwappedCmd, NetSerializeLayouts))
		{
			// 具体的添加到RepNotifies数组里
			RepNotifies->AddUnique(Parent.Property);
		}
	}
	// 如果没有OnRep方法，就直接序列化出属性就行了
	else
	{
		Cmd.Property->NetSerializeItem(Bunch, Bunch.PackageMap, Data + SwappedCmd);
	}
	// 有没有这个数组，应该肯定是有的
	if (GuidReferencesMap)
	{
		// 没有映射的GUID数量是否大于0，这个TrackedUnmappedGuids数组是在PackageMapClient::SerializeObject里面，如果客户端找不到Object就会添加到数组里
		const bool bHasUnmapped = TrackedUnmappedGuids.Num()> 0;
		// 这个if else的作用，看上去就是把 TrackedUnmappedGuids 里面的数据放到GuidReferencesMap这个里面，然后清空 TrackedUnmappedGuids 数组
		if (TrackedUnmappedGuids.Num() > 0 || TrackedDynamicMappedGuids.Num()> 0)
		{
			if (GuidReferences != nullptr)
			{
				if (!NetworkGuidSetsAreSame(GuidReferences->UnmappedGUIDs, TrackedUnmappedGuids))
				{
					bOutGuidsChanged = true;
				}
				else if (!NetworkGuidSetsAreSame(GuidReferences->MappedDynamicGUIDs, TrackedDynamicMappedGuids))
				{
					bOutGuidsChanged = true;
				}
			}
			if (GuidReferences == nullptr || bOutGuidsChanged)
			{
				GuidReferencesMap->Add(AbsOffset, FGuidReferences(Bunch, Mark, TrackedUnmappedGuids, TrackedDynamicMappedGuids, Cmd.ParentIndex, CmdIndex));
				bOutGuidsChanged = true;
			}
		}
		else
		{
			if (GuidReferences != nullptr)
			{
				GuidReferencesMap->Remove(AbsOffset);
				bOutGuidsChanged = true;
			}
		}
		// 清掉TrackedUnmappedGuids数组
		Bunch.PackageMap->ResetTrackedGuids(false);
		// 如果 TrackedUnmappedGuids 这个里面有值，就返回true
		return bHasUnmapped;
	}
	return false
}
```

```cpp
// 同步RPC属性相关
bool FObjectReplicator::ReceivedRPC(...)
{
	// 是否是服务器端
	const bool bIsServer = Connection->Driver->IsServer();
	// 获取属性复制器中的Object
	UObject* Object = GetObject();
	// 找到RPC的方法名称
	FName FunctionName = FieldCache->Field.GetFName();
	// 找到RPC方法的UFunction
	UFunction* Function = Object->FindFunction(FunctionName);
	// (!IsServer() || RepFlags.bNetOwner) 这是 ShouldCallRemoteFunction 这个方法的判断，客户端bCanExecute这个标志位肯定是true了
	const bool bCanExecute = Connection->Driver->ShouldCallRemoteFunction(Object, Function, RepFlags);
	if (bCanExecute)  
	{
		// bCanDelayRPC是false，所以bCanDelayUnmapped也是false
		bool bCanDelayUnmapped = bCanDelayRPC && Function->FunctionFlags & FUNC_NetReliable;
		// 先获取RPC方法的RepLayout
		TSharedPtr<FRepLayout> FuncRepLayout = Connection->Driver->GetFunctionRepLayout(LayoutFunction);
		// 客户端接收传过来的数据，
		FuncRepLayout->ReceivePropertiesForRPC(Object, LayoutFunction, OwningChannel, Reader, Parms, UnmappedGuids);
		if (bCanDelayUnmapped && (UnmappedGuids.Num() > 0 || PendingLocalRPCs.Num() > 0))
		{
			bOutDelayRPC = true;	
		}
		// 这个是我们自己加的，bOutDelayRPC这个标志位的意思就是客户端收到RPCBunch后不立即调用RPC方法
		else if(OwningChannel && OwningChannel->bDeferredRepNotify)
		{
			bOutDelayRPC = true;
		}
		else
		{
			// ActorChannel里面的Actor
			AActor* OwningActor = OwningChannel->Actor;
			// 这个ShouldForwardFunction方法，是个虚函数，如果是客户端返回true
			if (Connection->Driver->ShouldForwardFunction(OwningActor, Function, Parms))
			{
				// World上下文
				FWorldContext* const Context = GEngine->GetWorldContextFromWorld(Connection->Driver->GetWorld());
				if (Context != nullptr)
				{
					// 是子物体么
					UObject* const SubObject = Object != OwningChannel->Actor ? Object : nullptr;
					for (FNamedNetDriver& Driver : Context->ActiveNetDrivers)
					{
						// 看这个判断Driver.NetDriver != Connection->Driver，是说当前world中的NetDrive里面有不是Connection里的Driver么，如果有的话就调用一次ProcessRemoteFunction方法
						if (Driver.NetDriver != nullptr && (Driver.NetDriver != Connection->Driver) && Driver.NetDriver->ShouldReplicateFunction(OwningActor, Function))
						{
							Driver.NetDriver->ProcessRemoteFunction(OwningActor, Function, Parms, nullptr, nullptr, SubObject);
						}
					}
				}
			}
			// 客户端调用RPC方法
			Object->ProcessEvent(Function, Parms);
		}
	}
	return true;
}
```

```cpp
void UPackageMapClient::InternalWriteObject( FArchive & Ar, FNetworkGUID NetGUID, UObject* Object, FString ObjectPathName, UObject* ObjectOuter )
{
	// 
	/*
		GuidCache->ShouldAsyncLoad() 客户端是否支持异步加载Object
		IsNetGUIDAuthority() 是否在服务器上
		!GuidCache->IsExportingNetGUIDBunch 不能是ExportingNetGUIDBunch的流程
		!bNoLoad 客户端可以加载此Object
	*/
	// 满足条件后，会将Object的NetGUID添加到MustBeMappedGuidsInLastBunch这个数组里
	if ( GuidCache->ShouldAsyncLoad() && IsNetGUIDAuthority() && !GuidCache->IsExportingNetGUIDBunch && !bNoLoad )
	{
		MustBeMappedGuidsInLastBunch.AddUnique( NetGUID );
	}
	// 把NetGUID写入
	Ar << NetGUID;
	if ( NetGUID.IsDefault() )
	{
		ExportFlags.bHasPath = 1;
		Ar << ExportFlags.Value;
	}
	// 上边的ExportNetGUID这个方法就会走这里，如果我们Object通过ShouldSendFullPath这个方法，他就会写入bHasPath，意思是静态的Object,或者说是网络名称稳定的那种
	else if ( GuidCache->IsExportingNetGUIDBunch )
	{
		if ( Object != NULL )
		{
			ExportFlags.bHasPath = ShouldSendFullPath( Object, NetGUID ) ? 1 : 0;
		}
		else
		{
			ExportFlags.bHasPath = ObjectPathName.IsEmpty() ? 0 : 1;
		}
		ExportFlags.bNoLoad	= bNoLoad ? 1 : 0;
		Ar << ExportFlags.Value;
	}
	if ( ExportFlags.bHasPath )
	{
		if ( Object != NULL )
		{
			ObjectPathName = Object->GetName();
			ObjectOuter = Object->GetOuter();
		}
		// 是不是Package里面的Object，也就是场景中放置的Object
		// 如果Object不是网络名称稳定的，这个NetGUID.IsStatic()就返回true
		const bool bIsPackage = ( NetGUID.IsStatic() && Object != NULL && Object->GetOuter() == NULL );
		// 拿到Outer的GUID，然后递归调用
		FNetworkGUID OuterNetGUID = GuidCache->GetOrAssignNetGUID( ObjectOuter );
		InternalWriteObject( Ar, OuterNetGUID, ObjectOuter, TEXT( "" ), NULL );
		// 这部分就是写入ObjectPathName，这个就是Object那个路径吧
		if (Connection->Driver)
		{
			FName SearchPath = FName(*ObjectPathName);
			FName RenamedPath = Connection->Driver->RenamedStartupActors.FindRef(SearchPath);
			if (RenamedPath != NAME_None)
			{
				ObjectPathName = RenamedPath.ToString();
			}
		}
		// 这个GEnigne是走GameEngine么，打断点是走到的GameEngine
		GEngine->NetworkRemapPath(Connection, ObjectPathName, false);
		Ar << ObjectPathName;
		// 写入NetworkChecksum，还不知道这个干嘛的
		uint32 NetworkChecksum = 0;
		if ( ExportFlags.bHasNetworkChecksum )
		{
			NetworkChecksum = GuidCache->GetNetworkChecksum( Object );
			Ar << NetworkChecksum;
		}
		// 给ObjectLookup数组里存的，重新赋值
		FNetGuidCacheObject* CacheObject = GuidCache->ObjectLookup.Find( NetGUID );
		if ( CacheObject != NULL )
		{
			CacheObject->PathName			= FName( *ObjectPathName );
			CacheObject->OuterGUID			= OuterNetGUID;
			CacheObject->bNoLoad			= ExportFlags.bNoLoad;
			CacheObject->bIgnoreWhenMissing = ExportFlags.bNoLoad;
			CacheObject->NetworkChecksum	= NetworkChecksum;
		}
	}
}

// Object 这个Bunch里的Subobject，客户端需要序列化出来，然后传出去
// InternalLoadObjectRecursionCount 递归的次数，通过SerializeObject这里传进来是0
FNetworkGUID UPackageMapClient::InternalLoadObject( FArchive & Ar, UObject *& Object, const int InternalLoadObjectRecursionCount )  
{
	// 从Ar中读取出NetGUID
	FNetworkGUID NetGUID;  
	Ar << NetGUID;
	// 尝试通过NetGUID拿到他对应的Object，这个方法在客户端 子物体接收的地方看下
	if ( NetGUID.IsValid() && !NetGUID.IsDefault() )
	{
		Object = GetObjectFromNetGUID(NetGUID,GuidCache->IsExportingNetGUIDBunch );
	}
	// 和InternalWriteObject方法写入的时候保持一致，这里读取出FExportFlags
	FExportFlags ExportFlags;
	if ( NetGUID.IsDefault() || GuidCache->IsExportingNetGUIDBunch )
	{
		Ar << ExportFlags.Value;
	}
	// 记录我们的ImportNetGUID
	if ( GuidCache->IsExportingNetGUIDBunch )
	{
		if (NetGUID.IsValid())
		{
			GuidCache->ImportedNetGuids.Add( NetGUID );
		}
	}
	if ( ExportFlags.bHasPath )
	{
		// 递归，获取outer的NetGUID
		FNetworkGUID OuterGUID = InternalLoadObject( Ar, ObjOuter, InternalLoadObjectRecursionCount + 1 );
		// 读取我们的路径名称
		Ar << PathName;
		// 读取NetworkChecksum
		if ( ExportFlags.bHasNetworkChecksum )
		{
			Ar << NetworkChecksum;
		}
		// 网络名称稳定的NetGUID.IsStatic()为true
		const bool bIsPackage = NetGUID.IsStatic() && !OuterGUID.IsValid();
		
		GEngine->NetworkRemapPath( Connection, PathName, true );

		GuidCache->RegisterNetGUIDFromPath_Client( NetGUID, PathName, OuterGUID, NetworkChecksum, ExportFlags.bNoLoad, bIgnoreWhenMissing );
		// 再一次尝试通过NetGUID来拿到他对应的UObject
		Object = GuidCache->GetObjectFromNetGUID( NetGUID, GuidCache->IsExportingNetGUIDBunch );
	}
}
```

```cpp
// NetGUID，就是我们SubObject的NetGUID
// bIgnoreMustBeMapped，这个是GuidCache->IsExportingNetGUIDBunch标志位

UObject* FNetGUIDCache::GetObjectFromNetGUID( const FNetworkGUID& NetGUID, const bool bIgnoreMustBeMapped )
{
	// 在客户端拿到NetGUID对应的CacheObject
	FNetGuidCacheObject * CacheObjectPtr = ObjectLookup.Find( NetGUID );
	// 这个Objet就是SubObject
	UObject* Object = CacheObjectPtr->Object.Get();
	// 如果有SubObject了，就直接返回了
	if ( Object != NULL )
	{
		return Object;
	}
	// SubObject的Outer
	UObject* ObjOuter = NULL;
	// Outer的NetGUID是有效的
	if ( CacheObjectPtr->OuterGUID.IsValid() )
	{
		// 如果客户端这还没解析SubObject的Outer的话，这里提前返回
		FNetGuidCacheObject * OuterCacheObject = ObjectLookup.Find( CacheObjectPtr->OuterGUID );
		if ( OuterCacheObject == NULL )
		{
			return NULL;
		}
		if ( OuterCacheObject->bIsBroken )
		{
			CacheObjectPtr->bIsBroken = 1;
			return NULL;
		}
		// 根据Outer的NetGUID来解析出UObject
		ObjOuter = GetObjectFromNetGUID( CacheObjectPtr->OuterGUID, bIgnoreMustBeMapped );
		// 没解析出来,返回NULL
		if ( ObjOuter == NULL )
		{
			return NULL;
		}
	}
	// 在内存中找下SubObject是否存在
	Object = StaticFindObject( UObject::StaticClass(), ObjOuter, *CacheObjectPtr->PathName.ToString(), false );
	// 内存中没有，还没加载过Object。并且bNoLoad为false，也就是客户端可以加载这个SubObject（CanClientLoadObject这个方法决定bNoLoad标志位）
	if ( Object == NULL && !CacheObjectPtr->bNoLoad )
	{
		if ( bIsPackage )
		{
			// 如果开启了异步加载就无法立即求解出Object，而是开始异步加载流程
			if ( ShouldAsyncLoad() )
			{
				if (!PendingAsyncLoadRequests.Contains(CacheObjectPtr->PathName))
				{
					StartAsyncLoadingPackage(*CacheObjectPtr, NetGUID, false);
				}
				else
				{
					ValidateAsyncLoadingPackage(*CacheObjectPtr, NetGUID);
				}
				return NULL;
			}
			else
			{
				// 如果是Package就LoadPackage
				Object = LoadPackage( NULL, *CacheObjectPtr->PathName.ToString(), LOAD_None );
				SyncLoadedGUIDs.AddUnique(NetGUID);
			}
		}
		else
		{
			// 如果不是Package就LoadObject
			Object = StaticLoadObject( UObject::StaticClass(), ObjOuter, *CacheObjectPtr->PathName.ToString(), NULL, LOAD_NoWarn );
		}
	}
	// 缓存住我们加载出来的Object
	CacheObjectPtr->Object = Object;
	// 求解出来Object，我们给QueuedBunchObjectReferences数组里的重新写入
	UpdateQueuedBunchObjectReference(NetGUID, Object);
	
	return Object
}

		if (FQueuedBunchObjectReference* ObjectReference = SharedObjectReference.Get())
```

```cpp
	1. KeepProcessingActorChannelBunchesMap 这个数组的添加时在ActorChannel::CleanUp方法中，表示的含义就是我们在清理ActorChannel的时候，发现QueueBunch里面还有没来得及处理的Bunch，就添加到这个数组里面
	*/
	const bool bKeepProcessing = Connection->KeepProcessingActorChannelBunchesMap.Contains(ActorNetGUID);
	// 没有正在处理Bunch && (还有没来得及处理的Bunch || 时间限制，到时间了为true)
	if (bAllowProcessBunch && (bKeepProcessing || Connection->Driver->HasTimeToProcess()))
	{
		// 这个是我们自己添加的标志位，就是把整个ProcessBunch拆分了吧
		if (EnableProcessNewActorOnly)
		{
			// 没有生成过Actor，就SpawnedNewActor
			if (Actor == nullptr && QueuedBunches.Num())
			{
				FInBunch* QueuedInBunch = QueuedBunches[0];
				SpawnedNewActor(*QueuedInBunch);
			}
			// Actor生成错误，就清掉QueuedBunch
			if (!Actor)
			{
				QueuedBunches.Empty();
				return false;
			}
			// 继续处理Bunch
			while (QueuedBunches.Num())
			{
				FInBunch* QueuedInBunch = QueuedBunches[0];
				ProcessBunchWithoutNewActor(*QueuedInBunch);
				delete QueuedInBunch;
				QueuedBunches.RemoveAt(0);

				if (bSpawningNewActor)
				{
					PostSpawnedNewActor();
				}
			}
		}
		// 下面这个就是ue的流程，调用ProcessBunch方法来处理Bunch
		else
		{
			for (FInBunch* QueuedInBunch : QueuedBunches)
			{
				ProcessBunch(*QueuedInBunch);
				delete QueuedInBunch;
			}
			QueuedBunches.Empty();
		}
		// 调用Onrep方法吧
		for (auto& ReplicatorPair : ReplicationMap)
		{
			ReplicatorPair.Value->CallRepNotifies(true);
		}
	}
	else
	{
		QueuedBunchStartTime = FPlatformTime::Seconds();
	}
	return QueuedBunches.Num() == 0;
}
```

```cpp
void UNetDriver::ProcessRemoteFunctionForChannelPrivate(
	UActorChannel* Ch,// 调用RPC的Actor的ActorChannel
	const FClassNetCache* ClassCache,// TargetObject 的 ClassCache
	const FFieldNetCache* FieldCache,// RPC方法的FieldCache
	UObject* TargetObj,// 调用RPC的Object
	UNetConnection* Connection,// 调用RPC的Actor的NetConnection
	UFunction* Function,// RPC方法
	void* Parms,// Function的参数
	FOutParmRec* OutParms,
	FFrame* Stack,// 栈帧结构，蓝图用到吧我猜
	const bool bIsServer,// 是否是服务器端
	const ERemoteFunctionSendPolicy SendPolicy,//ERemoteFunctionSendPolicy::Default
	EProcessRemoteFunctionFlags& RemoteFunctionFlags)// None 默认的
{
	// 在服务器上OpenPacketId这个会在SendBunch里面记录，在客户端上会在ReceivedNextBunch里面记录，初始化的时候是INDEX_NONE
	if (Ch->OpenPacketId.First == INDEX_NONE)
	{
		if (!bIsServer)
		{
			return;
		}
		// 这个bIsReplicatingActor标志位是在ReplicateActor方法中设置为true
		if (Ch->bIsReplicatingActor)
		{
			return;
		}
		// 如果RemoteFunctionFlags标志位里不含有EProcessRemoteFunctionFlags::ReplicatedActor，就添加上
		if (!EnumHasAnyFlags(RemoteFunctionFlags, EProcessRemoteFunctionFlags::ReplicatedActor))
		{
			ReplicationFrame++;
			Ch->GetActor()->CallPreReplication(this);
			RemoteFunctionFlags |= EProcessRemoteFunctionFlags::ReplicatedActor;
		}
		// 最后调用ReplicateActor方法
		Ch->ReplicateActor();
	}
	// 如果当前ActorChannel无效了
	if (Ch->ChIndex == -1)
	{
		return;
	}
	// RepContext 上下文
	FScopedRepContext RepContext(Connection, Ch->GetActor());
	// 应该就是存储RPC的Bunch了吧
	FOutBunch Bunch(Ch, 0);
	// Function中是否带有FUNC_NetReliable标志位
	if (Function->FunctionFlags & FUNC_NetReliable)
	{
		Bunch.bReliable = 1;
	}
	// 如果栈帧为nullptr，会赋值LocalOutParms
	TArray<NetDriverInternal::FAutoDestructProperty> LocalOutParms;
	if (Stack == nullptr)
	{
		LocalOutParms = NetDriverInternal::CopyOutParametersToLocalParameters(Function, OutParms, Parms, TargetObj);
	}
	FNetBitWriter TempWriter( Bunch.PackageMap, 0 );
	// 拿到Function的RepLayout
	TSharedPtr<FRepLayout> RepLayout = GetFunctionRepLayout(Function);
	// SendProperties，将属性放到TempWriter里面
	RepLayout->SendPropertiesForRPC(Function, Ch, TempWriter, Parms);
	// 
	FNetFieldExportGroup* NetFieldExportGroup = Ch->GetOrCreateNetFieldExportGroupForClassNetCache(TargetObj);
	// 是否将Bunch放到Queue里面，意思就是立即发送Bunch还是放到Queue里面等待下一次Actor同步的时候一块发送。默认情况下，不稳定的多播都是放到Queue里面，其余的都是立即发送
	bool QueueBunch = false;
	switch (SendPolicy)
	{
		case ERemoteFunctionSendPolicy::Default:
			QueueBunch = ( !Bunch.bReliable && Function->FunctionFlags & FUNC_NetMulticast );
			break;
		case ERemoteFunctionSendPolicy::ForceQueue:
			QueueBunch = true;
			break;
		case ERemoteFunctionSendPolicy::ForceSend:
			QueueBunch = false;
			break;
	}
	// 放入到Queue里面的操作
	if (QueueBunch)
	{
		// 将TempWriter里的内容写进Bunch里面
		Ch->WriteFieldHeaderAndPayload(Bunch, ClassCache, FieldCache, NetFieldExportGroup, TempWriter);
		// 记录下大小吧
		ParameterBits = Bunch.GetNumBits();
	}
	// 立即发送的操作
	else
	{
		Ch->PrepareForRemoteFunction(TargetObj);
		// 局部TempBlockWriter Bunch
		FNetBitWriter TempBlockWriter(Bunch.PackageMap, 0);
		// 将TempWriter里的内容写进TempBlockWriter
		Ch->WriteFieldHeaderAndPayload(TempBlockWriter, ClassCache, FieldCache, NetFieldExportGroup, TempWriter);
		// 记录下TempWriter大小
		ParameterBits = TempBlockWriter.GetNumBits();
		// WriteContentBlockPayload方法，序列化TargetObj里的内容了，然后将TempBlockWriter在写入Bunch里面
		HeaderBits = Ch->WriteContentBlockPayload(TargetObj, Bunch, false, TempBlockWriter);
	}
	// 放入到Queue里面的操作
	if (QueueBunch)
	{
		// 下面看下这个方法
		Ch->QueueRemoteFunctionBunch(TargetObj, Function, Bunch);
	}
	// 立即发送的操作
	else
	{
		// 直接调用SendBunch了
		Ch->SendBunch(&Bunch, true);
	}
	// 如果是内部的就Flush
	if (Connection->IsInternalAck())
	{
		Connection->FlushNet();
	}
}
```
